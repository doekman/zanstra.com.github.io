Een Nederlandse postcode heeft het formaat: nNNNXX, waarbij n is 1 t/m 9 en N is 0 t/m 9 en X is A t/m Z (het getal van de postcode moet groter of gelijk aan 1000 zijn, m.a.w. niet met een 0 beginnen).
Er wordt geen onderscheid tussen hoofd en kleine letters gemaakt. 

function validatePostalCode(strPostalCode) {
  var rePostcode=new RegExp("[1-9][0-9]{3}[a-z]{2}","i");
  if(typeof(""+strPostalCode)=="string") {
    if(str.match(rePostcode)==str)
      alert('postcode voldoet');
    else
      alert('FOUTE postcode');
}
